- Install
git clone https://github.com/Microsoft/vcpkg.git
cd vcpkg
.\bootstrap-vcpkg.bat
git pull
./vcpkg update
./vcpkg install grpc:x64-windows
./vcpkg install protobuf:x64-windows
./vcpkg install boost:x64-windows
./vcpkg install openssl:x64-windows



- Compile proto files

- & "R:\C++\Projects\load-balancer\vcpkg\installed\x64-windows\tools\protobuf\protoc.exe" -I=. --cpp_out=proto proto/load_balancer.proto 

- & "R:\C++\Projects\load-balancer\vcpkg\installed\x64-windows\tools\protobuf\protoc.exe" --cpp_out=. --grpc_out=. --plugin=protoc-gen-grpc="R:\C++\Projects\load-balancer\vcpkg\installed\x64-windows\tools\grpc\grpc_cpp_plugin.exe" load_balancer.proto


- Build

- cmake -B build -S . -DCMAKE_BUILD_TYPE=Release -DCMAKE_TOOLCHAIN_FILE="R:/C++/Projects/load-balancer/vcpkg/scripts/buildsystems/vcpkg.cmake"
- cmake --build build --config Release


- Run
./load_balancer --backend-path ./backend_server --port 50050 --min-servers 3 --max-servers 6 --start-port 51000
